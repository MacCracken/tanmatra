1//! Error types for tanmatra.
2
3extern crate alloc;
4
5/// Errors that can occur in tanmatra computations.
6#[derive(Debug, Clone, PartialEq, serde::Serialize, serde::Deserialize, thiserror::Error)]
7#[non_exhaustive]
8pub enum TanmatraError {
9    /// Invalid element: atomic number out of range (1-118).
10    #[error("invalid element: Z={z} is out of range 1-118")]
11    InvalidElement {
12        /// The invalid atomic number.
13        z: u16,
14    },
15
16    /// Invalid isotope: mass number inconsistent with atomic number.
17    #[error("invalid isotope: Z={z}, A={a} (A must be >= Z and > 0)")]
18    InvalidIsotope {
19        /// Atomic number.
20        z: u16,
21        /// Mass number.
22        a: u16,
23    },
24
25    /// Invalid particle specification.
26    #[error("invalid particle: {reason}")]
27    InvalidParticle {
28        /// Description of why the particle is invalid.
29        reason: alloc::string::String,
30    },
31
32    /// Decay operation failed.
33    #[error("decay failed: {reason}")]
34    DecayFailed {
35        /// Description of why the decay failed.
36        reason: alloc::string::String,
37    },
38
39    /// Invalid energy value.
40    #[error("invalid energy: {reason}")]
41    InvalidEnergy {
42        /// Description of the energy error.
43        reason: alloc::string::String,
44    },
45
46    /// General computation error.
47    #[error("computation error: {reason}")]
48    ComputationError {
49        /// Description of the computation error.
50        reason: alloc::string::String,
51    },
52}
