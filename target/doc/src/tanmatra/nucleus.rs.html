1//! Nuclear structure: binding energy, mass defect, nuclear radius.
2//!
3//! Uses the Bethe-Weizsacker semi-empirical mass formula (liquid drop model)
4//! with standard coefficients.
5
6use crate::constants::R0_FM;
7use crate::error::TanmatraError;
8use serde::{Deserialize, Serialize};
9
10/// A nucleus characterized by atomic number Z and mass number A.
11#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
12pub struct Nucleus {
13    /// Atomic number (number of protons).
14    pub z: u16,
15    /// Mass number (number of protons + neutrons).
16    pub a: u16,
17}
18
19impl Nucleus {
20    /// Creates a new nucleus, validating that A >= Z and both are > 0.
21    ///
22    /// # Errors
23    ///
24    /// Returns [`TanmatraError::InvalidIsotope`] if A < Z or either is zero.
25    pub fn new(z: u16, a: u16) -> Result<Self, TanmatraError> {
26        if z == 0 || a == 0 || a < z {
27            return Err(TanmatraError::InvalidIsotope { z, a });
28        }
29        Ok(Self { z, a })
30    }
31
32    /// Number of neutrons (N = A - Z).
33    #[must_use]
34    #[inline]
35    pub fn n(&self) -> u16 {
36        self.a - self.z
37    }
38}
39
40// Bethe-Weizsacker coefficients (all in MeV).
41const A_V: f64 = 15.67;
42const A_S: f64 = 17.23;
43const A_C: f64 = 0.714;
44const A_A: f64 = 93.15;
45const DELTA_0: f64 = 12.0;
46
47/// Computes the nuclear binding energy (in `MeV`) using the Bethe-Weizsacker
48/// semi-empirical mass formula.
49///
50/// `B = a_v*A - a_s*A^(2/3) - a_c*Z*(Z-1)/A^(1/3) - a_a*(A-2Z)^2/(4A) + delta`
51///
52/// where delta = +12/sqrt(A) for even-even, -12/sqrt(A) for odd-odd, 0 otherwise.
53///
54/// # Errors
55///
56/// Returns [`TanmatraError::InvalidIsotope`] if Z or A is zero, or A < Z.
57pub fn binding_energy_mev(z: u16, a: u16) -> Result<f64, TanmatraError> {
58    if z == 0 || a == 0 || a < z {
59        return Err(TanmatraError::InvalidIsotope { z, a });
60    }
61
62    let af = f64::from(a);
63    let zf = f64::from(z);
64
65    let volume = A_V * af;
66    let surface = A_S * libm::pow(af, 2.0 / 3.0);
67    let coulomb = A_C * zf * (zf - 1.0) / libm::pow(af, 1.0 / 3.0);
68    let asymmetry = A_A * (af - 2.0 * zf) * (af - 2.0 * zf) / (4.0 * af);
69
70    let z_even = z.is_multiple_of(2);
71    let n_even = (a - z).is_multiple_of(2);
72    let delta = if z_even && n_even {
73        DELTA_0 / libm::sqrt(af)
74    } else if !z_even && !n_even {
75        -DELTA_0 / libm::sqrt(af)
76    } else {
77        0.0
78    };
79
80    let b = volume - surface - coulomb - asymmetry + delta;
81    Ok(b)
82}
83
84/// Binding energy per nucleon (in `MeV`).
85///
86/// # Errors
87///
88/// Returns [`TanmatraError::InvalidIsotope`] if Z or A is invalid.
89pub fn binding_energy_per_nucleon(z: u16, a: u16) -> Result<f64, TanmatraError> {
90    let b = binding_energy_mev(z, a)?;
91    Ok(b / f64::from(a))
92}
93
94/// Mass defect (in `MeV`): equals the binding energy in the semi-empirical model.
95///
96/// # Errors
97///
98/// Returns [`TanmatraError::InvalidIsotope`] if Z or A is invalid.
99pub fn mass_defect_mev(z: u16, a: u16) -> Result<f64, TanmatraError> {
100    binding_energy_mev(z, a)
101}
102
103/// Nuclear radius (in fm) using the empirical formula `R = R0 * A^(1/3)`.
104///
105/// # Errors
106///
107/// Returns [`TanmatraError::InvalidIsotope`] if A is zero.
108#[inline]
109pub fn nuclear_radius_fm(a: u16) -> Result<f64, TanmatraError> {
110    if a == 0 {
111        return Err(TanmatraError::InvalidIsotope { z: 0, a: 0 });
112    }
113    Ok(R0_FM * libm::cbrt(f64::from(a)))
114}

116/// Checks whether a number is a nuclear magic number.
117///
118/// Magic numbers correspond to completely filled nuclear shells and confer
119/// extra stability: 2, 8, 20, 28, 50, 82, 126.
120#[must_use]
121#[inline]
122pub fn is_magic_number(n: u16) -> bool {
123    matches!(n, 2 | 8 | 20 | 28 | 50 | 82 | 126)
124}

143#[cfg(test)]
144mod tests {
145    use super::*;
146
147    #[test]
148    fn nucleus_neutron_count() {
149        let u238 = Nucleus { z: 92, a: 238 };
150        assert_eq!(u238.n(), 146);
151    }
152
153    #[test]
154    fn invalid_nucleus_a_less_than_z() {
155        assert!(Nucleus::new(10, 5).is_err());
156    }
157
158    #[test]
159    fn invalid_nucleus_zero() {
160        assert!(Nucleus::new(0, 0).is_err());
161    }
162
163    #[test]
164    fn fe56_binding_energy_per_nucleon() {
165        let bea = binding_energy_per_nucleon(26, 56).unwrap();
166        assert!(bea > 8.5 && bea < 9.1, "Fe-56 B/A = {bea}");
167    }
168
169    #[test]
170    fn he4_binding_energy() {
171        let b = binding_energy_mev(2, 4).unwrap();
172        assert!(b > 24.0 && b < 32.0, "He-4 B = {b}");
173    }
174
175    #[test]
176    fn nuclear_radius_he4() {
177        let r = nuclear_radius_fm(4).unwrap();
178        assert!((r - 1.905).abs() < 0.01, "He-4 R = {r}");
179    }
180
181    #[test]
182    fn magic_numbers() {
183        let magic = [2, 8, 20, 28, 50, 82, 126];
184        for m in &magic {
185            assert!(is_magic_number(*m), "{m} should be magic");
186        }
187        assert!(!is_magic_number(3));
188        assert!(!is_magic_number(100));
189    }
190
191    #[test]
192    fn binding_energy_increases_with_a() {
193        let b12 = binding_energy_mev(6, 12).unwrap();
194        let b56 = binding_energy_mev(26, 56).unwrap();
195        assert!(b56 > b12);
196    }
197
198    #[test]
199    fn serde_roundtrip_nucleus() {
200        let n = Nucleus { z: 26, a: 56 };
201        let json = serde_json::to_string(&n).unwrap();
202        let back: Nucleus = serde_json::from_str(&json).unwrap();
203        assert_eq!(n, back);
204    }
